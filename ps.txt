# Lấy IP với fallback
try {
    $ipAddress = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notlike "*Loopback*" }).IPAddress[0]
} catch {
    $ipAddress = (Get-WmiObject Win32_NetworkAdapterConfiguration | Where-Object { $_.IPEnabled -and $_.IPAddress -ne $null }).IPAddress[0]
}

    content = "🚀 **Agent Debug Mode**"

        @{

            fields = @(
                @{ name = "Computer"; value = $env:COMPUTERNAME; inline = $true }
                @{ name = "User"; value = $env:USERNAME; inline = $true }
